X = user_record.values
sse = []  
k_range = range(1, 10)  # 尝试分1到9类

for k in k_range:
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    kmeans.fit(X)
    sse.append(kmeans.inertia_)#inertia欧式距离的平方和，衡量聚类的紧凑程度，k越大，他越小

plt.ylabel('SSE (Inertia)')
plt.title('Elbow Method for Optimal K')

def user_classif(data):
    #no为为购买过
    #new为新用户
    #unactive不活跃用户
    #active活跃用户
    #return回归用户
    status=[]
    for i in range(11):
        if data.iloc[i]==0:
            if len(status)>0:
                if status[i-1]=='no':
                    status.append('no')
                else:
                    status.append('unactive')
            else:
                status.append('no')
        else:
            if len(status)>0:
                if status[i-1]=='no':
                    status.append('new')
                elif status[i-1]=='unactive':
                    status.append('return')
                else :
                    status.append('active')
            else:
                status.append('new')
    return pd.Series(status,index=user_column)
